### 데이터베이스 설계
```sql
-- 사용자 테이블
users (id, name, email, role, department, created_at)

-- 출퇴근 기록
attendance (id, user_id, check_in, check_out, date, consecutive_days)

-- 건강 체크 기록 (강화된 데일리 설문)
health_checks (
    id, user_id, 
    condition,              -- 전반적 컨디션 (1-4)
    discomfort_areas,       -- 불편한 부위 (JSON 배열)
    wrist_pain,             -- 손목 통증 (1-10)
    sleep_hours,            -- 수면 시간
    fatigue_level,          -- 피로도 (1-10)
    work_motivation,        -- 작업 의욕 (1-10)
    date, created_at
)

-- 주간 설문
weekly_surveys (
    id, user_id, week_start_date,
    stress_level, stress_causes,
    physical_condition, mental_condition,
    submitted_at, reminder_sent_count
)

-- 월간 설문
monthly_surveys (
    id, user_id, month,
    overall_health, work_life_balance,
    improvement_suggestions,
    submitted_at, reminder_sent_count
)

-- 월간 종합 리포트 (자동 생성)
monthly_reports (
    id, user_id, month,
    avg_wrist_pain, avg_fatigue, avg_sleep_hours, avg_motivation,
    max_consecutive_days, total_work_days,
    health_score, recommendations,
    generated_at
)

-- 연속 출근 건강 예측
health_predictions (
    id, user_id, prediction_date,
    current_consecutive_days, current_health_score,
    predicted_score_7days, predicted_score_14days,
    optimal_work_pattern, created_at
)

-- 건강검진 관리
health_exams (id, user_id, exam_date, results, next_due_date)

-- 장비 지급 이력
equipment_history (id, user_id, equipment_type, issued_date, status)
```

CI/CD: GitHub Actions
